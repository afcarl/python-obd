11 -def doDir(path): 

12      # list all files in directory 
13      for f in os.listdir(path): 
14          # skip over '.', '..', and '.svn' 
15          if f == "." or f == ".." or f == ".svn": 
16              continue 
17          # either do the file or recurse 
18          if os.path.isfile(os.path.join(path, f)): 
19              sys.stderr.write("Setting props on " + os.path.join(path, f) + "... ") 
20              os.system(CMD + os.path.join(path, f)) 
21              sys.stderr.write(" OK.\n") 
22          elif os.path.isdir(os.path.join(path, f)): 
23              # recurse 
24              doDir(os.path.join(path, f))
